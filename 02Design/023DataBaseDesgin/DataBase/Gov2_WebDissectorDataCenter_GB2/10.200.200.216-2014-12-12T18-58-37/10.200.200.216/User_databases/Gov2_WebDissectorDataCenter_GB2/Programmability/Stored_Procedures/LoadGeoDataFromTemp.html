[dbo].[LoadGeoDataFromTemp]
MS_Description
加载地理信息从临时表哦
Properties
PropertyValue
ANSI Nulls OnTrue
Quoted Identifier OnTrue
SQL Script
CREATE PROCEDURE [dbo].[LoadGeoDataFromTemp]
AS
BEGIN
--抽取央网的数据，只取中国各省的数据
INSERT INTO dbo.DailyGeoVisits
       ( ProfileID ,
         VisitDate ,
         GeoKey ,
         Visits
       )
SELECT ProfileId,Date,GeoKey,Visits FROM
dbo.TempGeoVisitsStats s
LEFT JOIN dbo.DimGeography g ON s.CountryChineseName = g.CountryName AND s.StateProvinceChineseName = g.ProvinceName
WHERE GeoKey IS NOT NULL AND s.CountryChineseName = '中国' AND (g.LevelType = 2 OR g.GeoKey = 0)
AND profileid = 801026;

--抽取央网的其他国家的访问数据
INSERT INTO dbo.DailyGeoVisits
       (
         VisitDate ,
         GeoKey ,
         Visits
       )
SELECT Date,GeoKey,SUM(Visits) FROM
dbo.TempGeoVisitsStats s
INNER JOIN dbo.DimGeography g ON s.CountryChineseName = g.CountryName
WHERE GeoKey IS NOT NULL AND s.CountryChineseName != '中国' AND (g.LevelType = 1)
AND profileid = 801026
GROUP BY Date,GeoKey;


INSERT INTO dbo.DailyGeoVisits
       (
         VisitDate ,
         GeoKey ,
         Visits
       )
SELECT Date,GeoKey,SUM(Visits) FROM
dbo.TempGeoVisitsStats s
INNER JOIN dbo.DimGeography g ON s.CountryChineseName = g.CountryName
WHERE GeoKey IS NOT NULL AND (g.LevelType = 1 OR g.GeoKey = 0)
AND profileid != 801026
GROUP BY Date,GeoKey;

TRUNCATE TABLE dbo.TempGeoVisitsStats;
END;
GO
EXEC sp_addextendedproperty N'MS_Description', N'加载地理信息从临时表哦', 'SCHEMA', N'dbo', 'PROCEDURE', N'LoadGeoDataFromTemp', NULL, NULL
GO
Uses
  • [dbo].[DailyGeoVisits]
  • [dbo].[DimGeography]
  • [dbo].[TempGeoVisitsStats]
