--用于获取媒体协作模块的溯源分析数据
--@Legolas Li
CREATE PROCEDURE [dbo].[GetPublishTimeRanking]
@StartTime DATE,
@EndTime DATE,
@TopicKey INT
AS
BEGIN
SELECT TOP 30 MIN(MinPublishTime) AS MinTime,WebsiteName,ChannelName
FROM dbo.DailySocialMediaSynergyPerformance s
INNER JOIN dbo.Website w ON s.WebSiteKey = w.WebsiteKey
INNER JOIN dbo.ChannelWebsite c ON c.WebsiteKey = w.WebsiteKey
INNER JOIN dbo.Channel h ON h.ChannelKey = c.ChannelKey
WHERE s.Date BETWEEN @StartTime AND @EndTime
AND s.TopicKey = @TopicKey
GROUP BY WebsiteName,ChannelName
ORDER BY MinTime asc
END
GO
EXEC sp_addextendedproperty N'MS_Description', N'用于获取媒体协作模块的溯源分析数据', 'SCHEMA', N'dbo', 'PROCEDURE', N'GetPublishTimeRanking', NULL, NULL
GO
