CREATE View [dbo].[ProfileTopics] as

SELECT t.topickey,TopicName, w.KeyWordName from dimtopic t
inner join [dbo].[TopicKeywordRelation] R on t.topickey = r.topickey
left join [dbo].[DimKeyWords] w on w.keywordkey = r.KeywordKey

GO
