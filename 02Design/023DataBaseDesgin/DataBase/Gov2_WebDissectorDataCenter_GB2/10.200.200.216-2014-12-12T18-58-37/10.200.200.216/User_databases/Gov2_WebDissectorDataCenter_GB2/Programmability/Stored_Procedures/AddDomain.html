CREATE PROCEDURE [dbo].[AddDomain]
@domain nvarchar(128),
@domainName NVARCHAR(64),
@RootDomain NVARCHAR(64),
@WebSiteKey INT
as

DECLARE @domainKey INT;
SET @domainKey = -1;
SET @domainKey = ISNULL((SELECT DomainKey FROM dbo.Domain WHERE Domain = @domain AND RootDomain = @RootDomain),0);

IF (@domainKey <= 0)
BEGIN
INSERT INTO dbo.Domain
       (DomainName, Domain, RootDomain )
VALUES  (@domainName,@domain,@RootDomain)
SET @domainKey = (SELECT @@IDENTITY);
END;

IF NOT EXISTS(SELECT 1 FROM dbo.WebsiteDomain WHERE DomainKey = @domainKey AND WebsiteKey = @WebSiteKey)
BEGIN
   INSERT INTO dbo.WebsiteDomain
           ( WebsiteKey, DomainKey )
   VALUES  ( @WebSiteKey,@domainKey)
END;
GO
EXEC sp_addextendedproperty N'MS_Description', N'添加域名', 'SCHEMA', N'dbo', 'PROCEDURE', N'AddDomain', NULL, NULL
GO
EXEC sp_addextendedproperty N'MS_Description', N'域名', 'SCHEMA', N'dbo', 'PROCEDURE', N'AddDomain', 'PARAMETER', N'@domain'
GO
EXEC sp_addextendedproperty N'MS_Description', N'域的名字', 'SCHEMA', N'dbo', 'PROCEDURE', N'AddDomain', 'PARAMETER', N'@domainName'
GO
EXEC sp_addextendedproperty N'MS_Description', N'根域', 'SCHEMA', N'dbo', 'PROCEDURE', N'AddDomain', 'PARAMETER', N'@RootDomain'
GO
EXEC sp_addextendedproperty N'MS_Description', N'WebSite表主键', 'SCHEMA', N'dbo', 'PROCEDURE', N'AddDomain', 'PARAMETER', N'@WebSiteKey'
GO
