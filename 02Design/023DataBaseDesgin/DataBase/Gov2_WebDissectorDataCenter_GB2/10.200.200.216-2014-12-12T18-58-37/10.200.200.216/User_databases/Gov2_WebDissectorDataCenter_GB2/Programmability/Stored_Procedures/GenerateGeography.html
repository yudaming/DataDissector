[dbo].[GenerateGeography]
MS_Description
生成地理信息
Properties
PropertyValue
ANSI Nulls OnTrue
Quoted Identifier OnTrue
SQL Script
--Init Geo
CREATE procedure [dbo].[GenerateGeography]
as
INSERT INTO dbo.DimGeography
       ( GeoKey ,
           CountryName ,
           ProvinceName ,
           CityName ,
           CountryGeoCode ,
           ProvinceGeoCode ,
           CityGeoCode,
           LevelType
       )
SELECT LocationCode,L1Name,L2Name,L3Name
,CONVERT(BIGINT,L1Code),CONVERT(BIGINT,L2Code),CONVERT(BIGINT,L3Code),Type
FROM IPLocationV3.dbo.Location
WHERE type<=3
UNION
--添加对伊犁市的地理初始化，系IPLocationV3里的缺失记录
SELECT 1566500004100,'中国','新疆','伊犁市',156,156650000,1566500004100,3
UNION
SELECT 0,'-','-','-',0,0,0,0
GO
EXEC sp_addextendedproperty N'MS_Description', N'生成地理信息', 'SCHEMA', N'dbo', 'PROCEDURE', N'GenerateGeography', NULL, NULL
GO
