CREATE procedure [dbo].[GetGeoWeiboEmotions]
@topickey int,
@starttime DATE,
@endtime DATE  
as

select weightedsum/sumWeiboCount as emotionValue,provincename from (
select  provincename,sum(ReferedCount) as sumWeiboCount,
sum(cast(EmotionValue as bigint)*(cast(ReferedCount as bigint))) as weightedsum
from [dbo].DailyTopicEmotionsStats e
INNER join [dbo].[DimGeography] g on e.geocode = g.geokey
INNER join [dbo].[DimTopic] c on c.topickey = e.topickey
where c.topickey = @topickey and e.date between @starttime and @endtime
group by provincename
) t
GO
EXEC sp_addextendedproperty N'MS_Description', N'得到微博情绪统计和地理信息表', 'SCHEMA', N'dbo', 'PROCEDURE', N'GetGeoWeiboEmotions', NULL, NULL
GO
