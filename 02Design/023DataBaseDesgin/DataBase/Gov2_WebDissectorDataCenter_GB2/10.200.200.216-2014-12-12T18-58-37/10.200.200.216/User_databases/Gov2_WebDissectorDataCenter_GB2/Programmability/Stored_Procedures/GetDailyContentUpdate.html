/****** Script for SelectTopNRows command from SSMS  ******/
CREATE procedure [dbo].[GetDailyContentUpdate]
@StartDate date,
@EndDate date
as
--内容更新页面数据源
begin

SELECT p.ProfileID,c.UpdateDate,c.UpdateContent,p.ProfileName,L.LevelName,p.SupervisingAuthority
 FROM [dbo].[DailyContentUpdate] c
 left join Profile p on c.ProfileID = p.ProfileID
 left join ProfileLevel L on L.LevelID = c.ProfileLevelID
 where c.UpdateDate between @StartDate and @EndDate
 order by UpdateDate  desc
end
GO
EXEC sp_addextendedproperty N'MS_Description', N'得到每天的更新的内容', 'SCHEMA', N'dbo', 'PROCEDURE', N'GetDailyContentUpdate', NULL, NULL
GO
EXEC sp_addextendedproperty N'MS_Description', N'结束时间', 'SCHEMA', N'dbo', 'PROCEDURE', N'GetDailyContentUpdate', 'PARAMETER', N'@EndDate'
GO
EXEC sp_addextendedproperty N'MS_Description', N'开始时间', 'SCHEMA', N'dbo', 'PROCEDURE', N'GetDailyContentUpdate', 'PARAMETER', N'@StartDate'
GO
