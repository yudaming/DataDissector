[dbo].[InitProfileTimeWindow]
MS_Description
初始化profile时间窗口
Properties
PropertyValue
ANSI Nulls OnTrue
Quoted Identifier OnTrue
SQL Script
CREATE procedure [dbo].[InitProfileTimeWindow]
as
declare @datetime datetime;
set @datetime = dateadd(month,-2,getdate())
UPDATE [dbo].[ProfileTimeWindow]
  SET [ETLEndTime] = @datetime
     ,[FillDailySourceTypeStartTime] = @datetime
     ,[FillDailyKeywordsStartTime] = @datetime
     ,[ETLLastWeekEndTime] = @datetime
     ,[FillSummaryStartTime] = @datetime
     ,[FillDailyProfileNavigationStartTime] = @datetime
     ,[FillDailyStartTime] = @datetime
     ,[FillDailyChannelStartTime] = @datetime
     ,[FillDailyHotPageStartTime] = @datetime
     ,[FillDailyGeoStartTime] = @datetime
     ,[AvailableDailyEndTime] = @datetime
     ,[AvailableWeeklyEndTime] = @datetime
     ,[FillDailyGeoTitleStartTime] = @datetime
     ,[FillKeywordsUnusualTime] = @datetime
GO
EXEC sp_addextendedproperty N'MS_Description', N'初始化profile时间窗口', 'SCHEMA', N'dbo', 'PROCEDURE', N'InitProfileTimeWindow', NULL, NULL
GO
