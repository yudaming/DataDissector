CREATE procedure [dbo].[GetDailyWeiboEmotions]
@topickey int,
@starttime DATE,
@endtime DATE  
as

select weightedsum/sumWeiboCount as emotionValue,Date from (
select  Date,sum(ReferedCount) as sumWeiboCount,
sum(cast(EmotionValue as bigint)*(cast(ReferedCount as bigint))) as weightedsum
from [dbo].DailyTopicEmotionsStats e
INNER join [dbo].[DimGeography] g on e.geocode = g.geokey
INNER join [dbo].[DimTopic] c on c.topickey = e.topickey
where c.topickey = @topickey and e.date between @starttime and @endtime
group by Date
) t
GO
