CREATE PROCEDURE [dbo].[GetSummaryVisitorOfMonth]
   @startTime date,
   @endTime date,
   @profileID int
AS
--获取一段时间某一个porfile的访问量之和
BEGIN
 SELECT SUM(Visits) AS SumVisits FROM [dbo].[DailyGeoVisits] WHERE ProfileID=@profileID AND VisitDate BETWEEN @startTime AND @endTime
END
GO
EXEC sp_addextendedproperty N'MS_Description', N'得到每月访问量的概要', 'SCHEMA', N'dbo', 'PROCEDURE', N'GetSummaryVisitorOfMonth', NULL, NULL
GO
