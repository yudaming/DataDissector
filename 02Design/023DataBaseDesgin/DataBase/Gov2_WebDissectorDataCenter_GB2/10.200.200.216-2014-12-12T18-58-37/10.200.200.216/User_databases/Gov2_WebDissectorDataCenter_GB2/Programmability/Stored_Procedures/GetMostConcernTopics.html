--获取政策关注中政策关注度排名
CREATE PROCEDURE [dbo].[GetMostConcernTopics]
@BeginTime date,
@EndTime date
AS

SELECT t.TopicKey , t.TopicName
     ,SUM([PublishedTimes]) AS ReferedTimes
 FROM [dbo].[DailySocialMediaSynergyPerformance] d
 INNER JOIN dbo.DimTopic t ON d.TopicKey = t.TopicKey
 WHERE d.Date BETWEEN @BeginTime AND @EndTime
 GROUP BY t.TopicKey, TopicName
 ORDER BY ReferedTimes desc
GO
EXEC sp_addextendedproperty N'MS_Description', N'获取政策关注中政策关注度排名', 'SCHEMA', N'dbo', 'PROCEDURE', N'GetMostConcernTopics', NULL, NULL
GO
