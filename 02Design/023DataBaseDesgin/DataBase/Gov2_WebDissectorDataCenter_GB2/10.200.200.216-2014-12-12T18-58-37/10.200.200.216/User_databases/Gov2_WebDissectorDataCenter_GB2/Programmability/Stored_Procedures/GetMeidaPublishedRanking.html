CREATE PROCEDURE [dbo].[GetMeidaPublishedRanking]
@StartTime DATE,
@EndTime DATE
AS
BEGIN

SELECT SUM(PublishedTimes) AS SumPublishedTimes,WebsiteName
FROM dbo.DailySocialMediaSynergyPerformance s
LEFT JOIN dbo.Website w ON s.WebSiteKey = w.WebsiteKey
WHERE s.Date BETWEEN @StartTime AND @EndTime
GROUP BY WebsiteName
ORDER BY SumPublishedTimes DESC

END
GO
EXEC sp_addextendedproperty N'MS_Description', N'得到媒体发布的上级信息', 'SCHEMA', N'dbo', 'PROCEDURE', N'GetMeidaPublishedRanking', NULL, NULL
GO
