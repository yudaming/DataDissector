[dbo].[LoadDailyTopicAnalysisData]
MS_Description
加载每天的主题分析数据
Properties
PropertyValue
ANSI Nulls OnTrue
Quoted Identifier OnTrue
SQL Script
CREATE procedure [dbo].[LoadDailyTopicAnalysisData]
AS
INSERT INTO dbo.DailySocialMediaSynergyPerformance
       ( TopicKey ,
         WebSiteKey ,
         Date ,
         PublishedTimes ,
         ComentTimes ,
         MinPublishTime
       )
  select topickey,websitekey,time,0,[TopicReferTime],mindate from
          (SELECT dt.topickey
      ,w.websitekey
     ,td.[Domain],domainname,td.TIME,td.mindate
     ,td.[TopicReferTime],w.websitename
 FROM [dbo].[TempDailyTopicAnalysis] td
 INNER join domain m on m.rootdomain = td.domain
 INNER join [dbo].[WebsiteDomain] wd on wd.domainkey = m.domainkey
 INNER join [dbo].[Website] w on w.websitekey = wd.websitekey
 INNER join dbo.dimtopic dt on dt.topicname = td.topicname
 where w.websitekey is not null) s;
--truncate table [TempDailyTopicAnalysis];


with topicdocument as(
select topickey,sum(ComentTimes) as referedtimes,date from  DailySocialMediaSynergyPerformance
group by date,topickey)
INSERT INTO dbo.DailyTopicHotWordAnalysis
       ( TopicKey ,
         Date ,
         HotWords ,
         HotWordReferTimes ,
         DocumentReferTimes ,
         Correlation
       )
SELECT dt.topickey
     ,td.TIME
     ,[HotWords]
     ,[HotWordsReferTimes]
     ,isnull(referedtimes,0) as documentReferedTimes,ISNULL([Correlation],1)
FROM [dbo].[TempDailyTopicHotWordsAnalysis] td
INNER join dbo.dimtopic dt on dt.topicname = td.topicname
INNER join topicdocument d on d.topickey = dt.topickey AND td.Time = d.date
--truncate table [TempDailyTopicHotWordsAnalysis];
GO
EXEC sp_addextendedproperty N'MS_Description', N'加载每天的主题分析数据', 'SCHEMA', N'dbo', 'PROCEDURE', N'LoadDailyTopicAnalysisData', NULL, NULL
GO
Uses

  • [dbo].[TempDailyTopicAnalysis]
  • [dbo].[TempDailyTopicHotWordsAnalysis]
  • [dbo].[Website]
  • [dbo].[WebsiteDomain]
