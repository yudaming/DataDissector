 Create Procedure [dbo].[GetProfileDiffusionStats]
 @StartDate date,
 @EndDate date
 as
 --用于传播渠道的第2个页面即列表页面
  select avg([AvgPerformance]) as avgPerformance,L.LevelName,p.profilename,p.profileId,avg([SearchEnginePerformance]) as [SearchEnginePerformance] ,avg([OverseaPerformance]) as [OverseaPerformance]
     ,avg([GovSocialMediaPerformance]) as [GovSocialMediaPerformance]
     ,avg([MobileTerminalPerformance]) as [MobileTerminalPerformance]
     ,avg([SocialMediaSharePerformance]) as [SocialMediaSharePerformance] FROM [dbo].[DailyDiffusionStats] s
 left join Profile p on s.ProfileID = p.ProfileID
 left join ProfileLevel L on L.LevelID = s.ProfileLevelID
 where s.[DiffusionDate] between @StartDate and @EndDate
 group by L.LevelName,p.profilename,p.profileId
 order by avgPerformance desc
GO
EXEC sp_addextendedproperty N'MS_Description', N'得到profile扩散统计信息', 'SCHEMA', N'dbo', 'PROCEDURE', N'GetProfileDiffusionStats', NULL, NULL
GO
